# Starlit Citadel: essentially Citadel Uplink building turned into a zone, with job upkeep reduction to ease up on alloys
# essentially a soldier zone with job swapped for skywatchers. still 1 per planet limit cuz 200 range PD might be too much fun
zone_MZ_starlit_citadel = {
    icon = GFX_MZ_skywatcher
    base_buildtime = @zone_buildtime
	planet_limit = 1
    potential = { # planet scope
        days_passed > 720   # ensures you can't build it at start of the game when you don't have any event chains
        exists = owner
        owner = {
            has_origin = origin_starlit_citadel
            NOR = { # use this instead of country_flag = unlocks_warden_of_the_citadel since gestalts don't get that
                has_event_chain = abandoned_hatchery_chain_1
                has_event_chain = abandoned_hatchery_chain_2
                has_event_chain = abandoned_hatchery_chain_3
                has_event_chain = abandoned_hatchery_chain_4
            }
            is_ai = no						# AI isn't great at handling these
        }
    }

    unlock = { # planet scope
        days_passed > 720   # ensures you can't build it at start of the game when you don't have any event chains
        exists = owner
        owner = {
            has_origin = origin_starlit_citadel
            NOR = { # use this instead of country_flag = unlocks_warden_of_the_citadel since gestalts don't get that
                has_event_chain = abandoned_hatchery_chain_1
                has_event_chain = abandoned_hatchery_chain_2
                has_event_chain = abandoned_hatchery_chain_3
                has_event_chain = abandoned_hatchery_chain_4
            }
        }
		exists = solar_system
		solar_system = {
			custom_tooltip = {
				text = "requires_deep_space_citadel"
				count_ship_in_system = {
					limit = {
						is_space_citadel_ship_size = yes
					}
					count > 0
				}
			}
		}
    }

    resources = {
        category = planet_zones
        cost = {
            minerals = @zone_cost
        }
    }

    included_building_sets = {
        fortress
        automation
        origin
    }

    inline_script = {
        script = more_zones/zone_skywatchers_add
        AMOUNT = @scaling_district_1_job
    }

    # soldier upkeep reduction
	triggered_district_planet_modifier = {
		potential = {
            planet_is_ecu_equivalent = no
            planet_is_ring_world_equivalent = no
		}
		planet_soldiers_upkeep_mult = -0.01
	}
	triggered_district_planet_modifier = {
		potential = {
            planet_is_ecu_equivalent = yes
		}
		planet_soldiers_upkeep_mult = -0.02
	}
	triggered_district_planet_modifier = {
		potential = {
            planet_is_ring_world_equivalent = yes
		}
		planet_soldiers_upkeep_mult = -0.03
	}

    planet_modifier = {
        zone_building_slots_add = @BPV_ZONE_SLOT
    }

	zone_sets = {
		urban
		habitat_urban
	}

	inline_script = {
		script = zones/shared_city_non_urban_zone_modifiers
	}

	convert_to = {
		zone_fortress
	}
}

# Primal Calling: Wrangler jobs in a zone
# you can build it in habitats and ecus, but it incurs extra food upkeep
zone_MZ_primal_calling = {
    icon = GFX_MZ_wrangler
    base_buildtime = @zone_buildtime
    potential = { # planet scope
        exists = owner
        owner = {
            has_origin = origin_primal_calling
			has_encountered_every_fauna = yes
            is_ai = no						# AI isn't great at handling these
        }
    }

    unlock = { # planet scope
        exists = owner
        owner = {
            has_origin = origin_primal_calling
			has_encountered_every_fauna = yes
        }
    }

    resources = {
        category = planet_zones
        cost = {
            minerals = @zone_cost
        }
	}

	resources = {
		category = planet_districts_cities
		upkeep = {	# extra food upkeep on ecus
			food = 10
			mult = value:count_districts|TYPE|city|
			trigger = {
            	uses_district_set = city_world
			}
		}
		upkeep = {	# extra food upkeep on habitats
			food = 5
			mult = value:count_districts|TYPE|city|
			trigger = {
				planet_is_habitat_equivalent = yes
			}
		}
    }

    included_building_sets = {
        food
		urban
        origin
    }

    inline_script = {
        script = more_zones/zone_wranglers_add
        AMOUNT = @scaling_district_1_job
    }

    planet_modifier = {
        zone_building_slots_add = @BPV_ZONE_SLOT
    }

	zone_sets = {
		urban
		habitat_urban
	}

	inline_script = {
		script = zones/shared_city_non_urban_zone_modifiers
	}
}
