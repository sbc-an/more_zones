# medical: only available when genomic researcher is unlocked. built as an equivalent to hive spawning zones
## city
zone_MZ_medical = {
    inline_script = {
        script = more_zones/shared_medical_zone_effects
    }

    # regular planet
    triggered_district_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_regular_empire = yes
            }
            planet_is_habitat_equivalent = no
            planet_is_ecu_equivalent = no
            planet_is_ring_world_equivalent = no
        }
        job_healthcare_add = 15
    }

    # Habitats
    triggered_district_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_regular_empire = yes
            }
            planet_is_habitat_equivalent = yes
        }
        job_healthcare_add = 10 # don't think it's good idea to let people habitat spam pop print
        mult = owner.value:hab_void_dweller_jobs
    }

    # Ecus
    triggered_district_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_regular_empire = yes
            }
            planet_is_ecu_equivalent = yes
        }
        job_healthcare_add = 50
    }

    # Ringworlds
    triggered_district_planet_modifier = {
        potential = {
            exists = owner
            owner = {
                is_regular_empire = yes
            }
            planet_is_ring_world_equivalent = yes
        }
        job_healthcare_add = 100
    }

    zone_sets = {
        urban
		habitat_urban
        habitat_research
    }

	inline_script = {
		script = zones/shared_city_non_urban_zone_modifiers
	}
}

# politician: for those who uses politicians or its swaps
## city only. it doesn't make sense the ruling class would work away from the core of the city
zone_MZ_elite = {
    icon = GFX_MZ_strata_elite
    base_buildtime = @zone_buildtime

    resources = {
        category = planet_zones
        cost = {
            minerals = @zone_cost
        }
    }

    potential = { # planet scope
        hidden_trigger = { exists = owner }
        owner = {
			is_worker_coop_empire = no
			is_gestalt = no
            OR = {
                NOT = { has_planet_flag = toxic_knights_habitat }
                NOT = { owner = { has_origin = origin_toxic_knights } }
            }
            is_ai = no						# AI isn't great at handling these
        }
    }

    unlock = { # planet scope
        hidden_trigger = { exists = owner }
        owner = {
			is_worker_coop_empire = no
			is_gestalt = no
            OR = {
                NOT = { has_planet_flag = toxic_knights_habitat }
                NOT = { owner = { has_origin = origin_toxic_knights } }
            }
        }
    }

    inline_script = {
        script = more_zones/zone_politicians_add
		AMOUNT = @scaling_district_2_jobs
    }

    included_building_sets = {
        government
        origin
    }

    planet_modifier = {
        zone_building_slots_add = @BPV_ZONE_SLOT
    }

    zone_sets = {
        urban
		habitat_urban
    }

	convert_to = {
		zone_MZ_telepath
		zone_MZ_patrol
	}
}
